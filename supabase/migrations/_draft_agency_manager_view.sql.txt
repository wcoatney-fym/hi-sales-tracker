-- 1. Per-person manager identity (extend existing admin_credentials)
--    One row per manager (NOT one shared row per agency).
--    role='agency_admin' = manager; agency_id = the single agency (exclusivity).
ALTER TABLE admin_credentials
  ADD COLUMN IF NOT EXISTS agent_id     uuid REFERENCES agents(id) ON DELETE SET NULL,
  ADD COLUMN IF NOT EXISTS display_name text;
CREATE INDEX IF NOT EXISTS idx_admin_credentials_agency ON admin_credentials(agency_id);
CREATE INDEX IF NOT EXISTS idx_admin_credentials_agent  ON admin_credentials(agent_id);
-- Password log: global-admin reads admin_credentials WHERE role='agency_admin';
-- agency-admin reads only its own agency_id rows (enforced in admin-api).

-- 2. Two-way nudge thread (extend existing at_risk_activities)
ALTER TABLE at_risk_activities
  ADD COLUMN IF NOT EXISTS author_role text CHECK (author_role IN ('manager','agent')),
  ADD COLUMN IF NOT EXISTS kind        text NOT NULL DEFAULT 'note'
                                       CHECK (kind IN ('note','nudge','flag'));

-- 3. Manager-only disposition
CREATE TABLE IF NOT EXISTS policy_dispositions (
  id            uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  policy_id     uuid NOT NULL REFERENCES form_submissions(id) ON DELETE CASCADE,
  agency_id     uuid NOT NULL REFERENCES agencies(id) ON DELETE CASCADE,
  disposition   text NOT NULL CHECK (disposition IN ('working','secured','lost','follow_up')),
  note          text NOT NULL DEFAULT '',
  set_by        uuid,
  set_at        timestamptz NOT NULL DEFAULT now(),
  CONSTRAINT policy_dispositions_unique UNIQUE (policy_id)
);
CREATE INDEX IF NOT EXISTS idx_policy_dispositions_agency ON policy_dispositions(agency_id);

-- 4. Notifications (drives the agent login popup)
CREATE TABLE IF NOT EXISTS notifications (
  id            uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  recipient_id  uuid NOT NULL,
  agency_id     uuid NOT NULL REFERENCES agencies(id) ON DELETE CASCADE,
  policy_id     uuid REFERENCES form_submissions(id) ON DELETE CASCADE,
  type          text NOT NULL CHECK (type IN ('nudge','flag','reply')),
  read_at       timestamptz,
  created_at    timestamptz NOT NULL DEFAULT now()
);
CREATE INDEX IF NOT EXISTS idx_notifications_recipient_unread
  ON notifications(recipient_id) WHERE read_at IS NULL;

-- RLS: every table stays locked to service_role (access via edge functions),
-- consistent with the current architecture. Role gating (manager-only
-- disposition writes) is enforced in admin-api / agent-webhook.
